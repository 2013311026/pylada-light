__docformat__ = "restructuredtext en"
__all__ = ['bcc', 'fcc']


[docs]def bcc():
  """ Creates a BCC lattice with a single site. """
  from pylada.crystal import Structure
  return Structure( -0.5, 0.5, 0.5,\
                    0.5, -0.5, 0.5,\
                    0.5, 0.5, -0.5,\
                    scale=1, name='bcc' )\
           .add_atom(0, 0, 0, 'A')
